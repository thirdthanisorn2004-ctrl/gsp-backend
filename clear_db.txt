import sqlite3
import os

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
DB_PATH = os.path.join(BASE_DIR, "relay_data.db")

conn = sqlite3.connect(DB_PATH)
cursor = conn.cursor()

cursor.execute("DROP TABLE IF EXISTS relays")

# สร้างตารางใหม่ กำหนดชนิดข้อมูลให้ชัดเจน (REAL = ค่าตัวเลขจุดทศนิยม)
cursor.execute("""
CREATE TABLE relays (
    Plant TEXT,
    Breaker TEXT,
    CT_Ratio TEXT,
    CT_Class TEXT,
    CT_Burden_VA TEXT,
    ZCT_Ratio TEXT,
    ZCT_Class TEXT,
    ZCT_Burden_VA TEXT,
    Relay_ID TEXT UNIQUE, 
    Relay_Manufacturer TEXT,
    Relay_Model TEXT,
    Phase_Curve TEXT,
    Phase_Pickup REAL,
    Phase_Prim_Amps REAL,
    Phase_Time_Dial REAL,
    Phase_Inst_Pickup REAL,
    Phase_Inst_Prim_Amps REAL,
    Phase_Inst_Delay_s REAL,
    Ground_Curve TEXT,
    Ground_Pickup REAL,
    Ground_Prim_Amps REAL,
    Ground_Time_Dial REAL,
    Ground_Inst_Pickup REAL,
    Ground_Inst_Prim_Amps REAL,
    Ground_Inst_Delay_s REAL,
    OLR_Trip REAL,
    OLR_Prim_Amps REAL,
    OLR_Time_Constant REAL,
    Remark TEXT
)
""")

# ยัดข้อมูล 28 แถว โดยใส่แค่ Plant กับ Relay_ID นอกนั้นปล่อยเป็น NULL อัตโนมัติ
for i in range(1, 29):
    # สร้าง Relay_ID แบบคงที่ เช่น GSP1-R01, GSP1-R02 ไปจนถึง R28
    relay_id = f"GSP1-R{i:02d}"
    cursor.execute("""
    INSERT INTO relays (Plant, Relay_ID) VALUES (?, ?)
    """, ('GSP1', relay_id))

conn.commit()
print("อัปเกรดฐานข้อมูลเรียบร้อย! คอลัมน์ตัวเลขเป็น REAL และใช้ NULL แทนขีดแล้ว")
conn.close()
